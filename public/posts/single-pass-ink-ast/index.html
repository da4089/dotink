reduce := (list, f, acc) => (
	(reducesub := (idx, acc) => (
		idx :: {
			len(list) -> acc
			_ -> reducesub(
				idx + 1
				f(acc, list.(idx))
			)
		}
	)
	)(0, acc)
)
